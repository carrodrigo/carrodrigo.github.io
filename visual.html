# Agrupar por semana y calcular las ventas totales por semana
ventas_semanales <- datos_limpios %>%
  group_by(Week = floor_date(InvoiceDate, "week")) %>%
  summarise(TotalSales = sum(Sales))

# Graficar la serie de tiempo de ventas semanales con Plotly
grafico_ventas_semanales_plotly <- plot_ly(ventas_semanales, x = ~Week, y = ~TotalSales, type = 'scatter', mode = 'lines') %>%
  layout(title = "Ventas Semanales (Sin Cancelaciones) a lo Largo del Tiempo",
         xaxis = list(title = "Semana"),
         yaxis = list(title = "Ventas Totales"))

print(grafico_ventas_semanales_plotly)
# Agrupar por mes y calcular las ventas totales por mes
ventas_mensuales <- datos_limpios %>%
  group_by(Month = floor_date(InvoiceDate, "month")) %>%
  summarise(TotalSales = sum(Sales))

# Graficar la serie de tiempo de ventas mensuales con Plotly
grafico_ventas_mensuales_plotly <- plot_ly(ventas_mensuales, x = ~Month, y = ~TotalSales, type = 'scatter', mode = 'lines') %>%
  layout(title = "Ventas Mensuales (Sin Cancelaciones) a lo Largo del Tiempo",
         xaxis = list(title = "Mes"),
         yaxis = list(title = "Ventas Totales"))

print(grafico_ventas_mensuales_plotly)
# Adicionalmente, para identificar patrones inusuales, graficamos con media móvil usando Plotly

# Calcular la media móvil de 7 días
ventas_diarias <- ventas_diarias %>%
  mutate(SMA_7 = rollmean(TotalSales, k = 7, fill = NA, align = "center"))

grafico_ventas_diarias_sma_plotly <- plot_ly(ventas_diarias, x = ~Date) %>%
  add_trace(y = ~TotalSales, name = 'Ventas Diarias', mode = 'lines',type = 'scatter') %>%
  add_trace(y = ~SMA_7, name = 'Media Móvil (7 días)', mode = 'lines',type = 'scatter') %>%
  layout(title = "Ventas Diarias con Media Móvil (Sin Cancelaciones)",
         xaxis = list(title = "Fecha"),
         yaxis = list(title = "Ventas"),
         legend = list(orientation = "h", yanchor = "bottom", y = -0.2, xanchor = "left", x = 0))

print(grafico_ventas_diarias_sma_plotly)
